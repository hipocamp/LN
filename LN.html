    <script>

        let targetValue = 0;

        let gameActive = true;

        let results = []; // Aquí guardarem tots els intents


        function initGame() {

            gameActive = true;

            targetValue = Math.floor(Math.random() * 101);

            document.getElementById('target-number').innerText = targetValue;

            document.getElementById('user-marker').classList.add('hidden');

            document.getElementById('correct-marker').classList.add('hidden');

            document.getElementById('feedback').classList.add('opacity-0');

        }


        document.getElementById('click-area').addEventListener('pointerdown', function(e) {

            if (!gameActive) return;

            gameActive = false;


            const rect = this.getBoundingClientRect();

            const x = e.clientX - rect.left;

            const percentage = Math.max(0, Math.min(1, x / rect.width));

            const userGuess = Math.round(percentage * 100);

            const error = Math.abs(targetValue - userGuess);


            // Guardar dades de l'intent

            results.push({

                intent: results.length + 1,

                objectiu: targetValue,

                resposta: userGuess,

                error: error,

                data: new Date().toLocaleTimeString()

            });


            document.getElementById('trials-count').innerText = "Intents: " + results.length;

            

            showResults(userGuess, percentage, error);

        });


        function showResults(guess, percentage, error) {

            const userMarker = document.getElementById('user-marker');

            const correctMarker = document.getElementById('correct-marker');

            

            userMarker.style.left = (percentage * 100) + "%";

            userMarker.classList.remove('hidden');

            

            correctMarker.style.left = (targetValue) + "%";

            

            setTimeout(() => {

                correctMarker.classList.remove('hidden');

                document.getElementById('feedback').classList.remove('opacity-0');

                document.getElementById('result-text').innerText = `Has marcat ${guess} (Error: ${error})`;

            }, 300);

        }


        function nextGame() { initGame(); }


        // --- FUNCIONS DE DADES ---


        function generateCSV() {

            let csv = "Intent,Objectiu,Resposta,Error,Hora\n";

            results.forEach(r => {

                csv += `${r.intent},${r.objectiu},${r.resposta},${r.error},${r.data}\n`;

            });

            return csv;

        }


        function downloadResults() {

            if (results.length === 0) return alert("No hi ha dades.");

            const csv = generateCSV();

            const blob = new Blob([csv], { type: 'text/csv' });

            const url = window.URL.createObjectURL(blob);

            const a = document.createElement('a');

            a.setAttribute('hidden', '');

            a.setAttribute('href', url);

            a.setAttribute('download', 'resultats_estimacio.csv');

            document.body.appendChild(a);

            a.click();

            document.body.removeChild(a);

        }


        function sendByEmail() {

            if (results.length === 0) return alert("No hi ha dades.");

            

            const email = "exemple@correu.com"; // CANVIA AIXÒ pel teu email

            const subject = "Resultats Estimació Numèrica - NeurekaLAB";

            const body = "Aquí tens els resultats de la sessió:\n\n" + generateCSV();

            

            // Això obre el gestor de correu del mòbil/PC amb el text ja posat

            window.location.href = `mailto:${email}?subject=${encodeURIComponent(subject)}&body=${encodeURIComponent(body)}`;

        }


        window.onload = initGame;
